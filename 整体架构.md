# 代码整体架构
rescue_opt_agent/
├── app/
│   ├── cli.py                         # 终端入口
│   ├── api.py                         # 可选 FastAPI 入口
│   └── session.py                     # 会话初始化、session_id 管理
├── config/
│   ├── settings.py                    # 模型、路径、求解器、数据库配置
│   └── logging.py                     # 日志配置
├── llm/
│   ├── client.py                      # OpenAI 官方 API 封装，Responses API/Chat 调用统一出口
│   └── prompts/
│       ├── system_prompt.md
│       ├── confirm_prompt.md
│       ├── codegen_prompt.md
│       └── repair_prompt.md
├── core/
│   ├── agent.py                       # 智能体基类
│   ├── react_agent.py                 # ReAct_agent实现，主循环：Thought/Action/Observation
│   ├── router.py                      # 判断当前走确认、检索、生成、修复哪条分支
│   ├── intent_parser.py               # 识别问题类型，对应 skill
│   ├── problem_builder.py             # 汇总用户输入 + skill + rag，形成结构化问题
│   ├── confirmation.py                # 生成“待确认问题说明”
│   ├── recovery.py                    # 代码报错后的修复与再规划
│   └── tool_calling.py                # tool schema 注册与调用分发
├── schemas/
│   ├── problem.py                     # 问题定义 schema
│   ├── constraint.py                  # 约束 schema
│   ├── objective.py                   # 目标函数 schema
│   ├── tool_io.py                     # 各工具输入输出 schema
│   ├── solution.py                    # 求解结果 schema
│   └── memory.py                      # 记忆条目 schema
├── skills/
│   ├── emergency_island_supply/
│   │   ├── skill.md                   # 问题类型说明、目标函数、基础约束
│   │   ├── examples.md                # 示例输入/输出
│   │   └── metadata.json              # skill 名称、标签、关键词、适用条件
│   ├── vehicle_routing/
│   │   ├── skill.md
│   │   ├── examples.md
│   │   └── metadata.json
│   └── loader.py                      # skill 读取、检索、匹配
├── knowledge/
│   ├── retrievers/
│   │   ├── skill_retriever.py         # skill 检索
│   │   ├── resource_retriever.py      # 车辆、仓库、库存、时窗等资源检索
│   │   └── experience_retriever.py    # 历史经验检索
│   ├── ingestors/
│   │   ├── resource_ingestor.py
│   │   └── experience_ingestor.py
│   └── indexes/
│       ├── resources/                 # 向量库/索引文件
│       └── experiences/
├── memory/
│   ├── short_term.py                  # 当前会话记忆
│   ├── long_term.py                   # 历史经验记忆
│   ├── summarizer.py                  # 会话压缩与经验提炼
│   └── store.py                       # SQLite / 本地文件 / 向量库统一接口
├── tools/
│   ├── registry.py                    # 可调用工具注册中心
│   ├── heuristic_solver_tool.py       # 启发式算法代码生成 + 执行入口
│   ├── exact_solver_tool.py           # 精确求解代码生成 + 执行入口
│   ├── code_repair_tool.py            # 根据报错修复代码
│   ├── sandbox_exec_tool.py           # 隔离执行
│   ├── validator_tool.py              # 校验约束、数据完整性
│   └── visualization_tool.py          # 路线图/甘特图等
├── solver/
│   ├── codegen/
│   │   ├── heuristic_generator.py     # 生成启发式算法代码
│   │   ├── exact_generator.py         # 生成 MILP/CP 等精确算法代码
│   │   └── templates/                 # 通用代码模板
│   ├── sandbox/
│   │   ├── executor.py                # 代码执行器
│   │   ├── runner.py                  # 执行流程控制
│   │   └── capture.py                 # stdout/stderr/result 抓取
│   └── postprocess/
│       ├── parser.py                  # 解析求解输出
│       └── evaluator.py               # 解质量评估
├── storage/
│   ├── db/
│   │   └── agent.db
│   ├── runs/
│   │   ├── generated_code/
│   │   ├── results/
│   │   └── plots/
│   └── cache/
├── tests/
│   ├── test_intent_parser.py
│   ├── test_problem_builder.py
│   ├── test_react_loop.py
│   └── test_tools.py
├── main.py                            # 简单启动入口，可转调 app/cli.py
├── requirements.txt
├── .env.example
└── Dockerfile
